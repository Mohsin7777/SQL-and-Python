

PROC SQL;

CREATE TABLE CT_SUM AS

SELECT BILL_YEAR, BILL_MONTH, BAN, SUBSCRIBER_NO

, SUM(CASE When RLH=1  Then ACTV_AMT Else 0 End) AS  CT_RLH_CHARGES

, SUM(CASE When RLH=0  Then ACTV_AMT Else 0 End) AS  CT_NONRLH_CHARGES

FROM  DATAETE.TEMP_CT

GROUP BY BILL_YEAR, BILL_MONTH, BAN, SUBSCRIBER_NO, RLH


;QUIT;
