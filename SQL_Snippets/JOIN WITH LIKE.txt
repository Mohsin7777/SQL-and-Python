

SELECT A.BAN, A.SUBSCRIBER_NO, A.FRANCHISE_ELIGIBILITY, A.CHANNEL_ID, A.EFFECTIVE_DATE, A.EXPIRATION_DATE, A.TO_SOC, A.TO_SOC_DESC, A.TO_RATE, A.GROSS_ADD_MIGRATION_FROM_SOC, A.FROM_SOC_DESCRIPTION, A.FROM_RATE
, B.MEMO_SYSTEM_TXT , B.MEMO_DATE, B.MEMO_SUBSCRIBER, B.MEMO_BAN, B.MEMO_USER_ID
INTO #PP_CSM
FROM #MIGRATIONS_BY_RATE_POSTPAID_WIRELESS_BANCTN A
INNER JOIN STAGE.dbo.STG_CSM_MEMO B
ON (A.BAN=B.MEMO_BAN OR A.SUBSCRIBER_NO=B.MEMO_SUBSCRIBER)
INNER JOIN #MEMOS C
ON B.MEMO_TYPE=C.MEMTP_MEMO_TYPE
WHERE B.MEMO_SYSTEM_TXT LIKE  '%'+CONVERT(VARCHAR(10), a.EFFECTIVE_DATE, 101) +'%'
